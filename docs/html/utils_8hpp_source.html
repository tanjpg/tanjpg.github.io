120 inline bool parse_pose_vector(
121  const std::vector<double> & param,
122  geometry_msgs::msg::Pose & pose)
123 {
124  // Location is x y z qx qy qz qw
125  if (param.size() == 7) {
126  pose.position.x = param[0];
127  pose.position.y = param[1];
128  pose.position.z = param[2];
129  tf2::Quaternion temp_qr(
130  param[3],
131  param[4],
132  param[5],
133  param[6]);
134  pose.orientation = tf2::toMsg(temp_qr.normalized());
135  return true;
136  } else if (param.size() == 6) {
137  // Location is x y z roll pitch yaw
138  pose.position.x = param[0];
139  pose.position.y = param[1];
140  pose.position.z = param[2];
141  tf2::Quaternion temp_qr;
142  temp_qr.setRPY(
143  param[3],
144  param[4],
145  param[5]);
146  pose.orientation = tf2::toMsg(temp_qr);
147  return true;
148  } else {
149  // Location invalid
150  return false;
151  }
152 }

159 inline void declare_or_get_param(
160  T & output_value,
161  const std::string & param_name,
162  const rclcpp::Node::SharedPtr & node,
163  const rclcpp::Logger & logger,
164  const T & default_value = T{})
165 {
166  try {
167  if (node->has_parameter(param_name)) {
168  node->get_parameter_or<T>(param_name, output_value, default_value);
169  } else {
170  output_value = node->declare_parameter<T>(param_name, default_value);
171  }
172  } catch (const rclcpp::exceptions::InvalidParameterTypeException & e) {
173  // Catch a <double> parameter written in the yaml as "1" being considered an <int>
174  if (std::is_same<T, double>::value) {
175  node->undeclare_parameter(param_name);
176  output_value = node->declare_parameter<int>(param_name, 0);
177  } else {
178  RCLCPP_ERROR(
179  logger,
180  "Error getting parameter \'%s\', check parameter type in YAML file.",
181  param_name.c_str());
182  throw e;
183  }
184  }
185  RCLCPP_INFO_STREAM(
186  logger,
187  "Found parameter - " << param_name << ": " << output_value);
188 }
