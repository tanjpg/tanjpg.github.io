143 static inline geometry_msgs::msg::Pose poseOctomapToTf(
144  const octomap::pose6d & octomapPose)
145 {
146  auto r = quaternionOctomapToTf(octomapPose.rot());
147  geometry_msgs::msg::Quaternion orientation;
148  orientation.x = r.x();
149  orientation.y = r.y();
150  orientation.z = r.z();
151  orientation.w = r.w();
152 
153  geometry_msgs::msg::Pose pose;
154  pose.position = pointOctomapToTf(octomapPose.trans());
155  pose.orientation = orientation;
156  return pose;
157 }
